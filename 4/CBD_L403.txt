load csv with Headers from 'file:///git_selection.csv' as row
MERGE (n:User {svn_id:row.svn_id, name: row.real_name})
merge (m:Project {name : row.project_name })
merge (n) -[:ROLE {name: row.role_on_project}]-> (m)
Merge (l:Organization {name: row.organization})
merge (m) -[:AT ]-> (l)

c)
#1
MATCH (n:User) 
RETURN n

#2
MATCH (n:User) 
RETURN n.name AS Nome

#3
MATCH (n:Project)<-[:ROLE]-(m:User) 
RETURN distinct n.name AS Projeto

#4
MATCH (n:Project)<-[:ROLE]-(m:User) 
RETURN m.name AS Nome, count(n) AS Total

#5
MATCH (n:Project)<-[:ROLE]-(m:User) 
RETURN m.name AS Nome, count(n) AS Total
ORDER BY Total DESC

#6
MATCH (n:Project)<-[:ROLE]-(m:User) 
RETURN n.name AS Nome, count(m) AS Total
ORDER BY Total DESC

#7
MATCH (n:Project)<-[:ROLE {name : "Committer"}]-(m:User) 
RETURN n.name AS Nome, count(m) AS TotalCommiters
ORDER BY TotalCommiters DESC

#8
MATCH (p:User)-[:ROLE]->(n:Project)<-[:ROLE]-(m:User {svn_id  : "atm"}) 
RETURN m.name AS NomeATM, p.name AS NomeUser2, n.name AS Projeto

#9
MATCH (p:User)-[:ROLE {name : "Committer"}]->(n:Project)<-[:ROLE {name : "PMC"}]-(m:User {svn_id  : "atm"}) 
RETURN m.name AS NomeATM, p.name AS NomeUser2, n.name AS Projeto
